    2Advance

    Advance while developing on this App. Learn how to use proper platform tags in your Definitions and how to integrate your Definitions with the lychee.pkg file. 
    
lychee.define('app.entity.Custom').tags({
	platform: 'html'
}).supports(function(lychee, global) {

	if (typeof global.ArrayBuffer === 'function') {
		return true;
	}

	return false;

}).exports(function(lychee, app, global, attachments) {

	var _ArrayBuffer = global.ArrayBuffer;

	var Class = function() {

		this.buffer = new _ArrayBuffer();

	};

	Class.prototype = {

	};

	return Class;

});

    3Migrate

    When you're ready, migrate this App into a lycheeJS Integrated App with full isomorphic capabilities. 
    
cd ./my-project;

lycheejs-breeder migrate;
    Always remember, don't use platform-specific code. HTML and CSS are platform-specific, too!
